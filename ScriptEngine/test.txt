int sum(int arg0, int arg1)
{
	return arg0 + arg1
}

float fsum(float f0, float f1)
{
	return f0 + f1
}

float castTest(int arg0, float arg1)
{
	return arg0 + arg1
}

type Vec2
{
	float x
	float y
}

float memberAccess(Vec2& _vec)
{
	return _vec.y
}

int precedence(int a, int b)
{
	return 3 + a * (b + 3) * a + b
}
